加密方式是出题人魔改的rc4

出题人的思路是查看左侧有rc4init函数直接套用题目模板的魔改rc4，以题目中的one piece数组作为密文反解出实际的flag

解密代码如下

```c
#include <stdio.h>
#include <string.h>
int main() {
    unsigned char data[] = {
        0xBB, 0xCA, 0x12, 0x14, 0xD0, 0xF1, 0x99, 0xA7, 0x91, 0x48,
        0xC3, 0x28, 0x73, 0xAD, 0xB7, 0x75, 0x8C, 0x89, 0xCD, 0xDD,
        0x2D, 0x50, 0x5D, 0x7F, 0x95, 0xB1, 0xA4, 0x9D, 0x09, 0x43,
        0xE1, 0xD2, 0xE9, 0x66, 0xEA, 0x18, 0x98, 0xC6, 0xCC, 0x02,
        0x39, 0x18
    };
    unsigned char real_key[] = "EC3700DFCD4F364EC54B19C5E7E26DEF6A25087C4FCDF4F8507A40A9019E3B48BD70129D0141A5B8F089F280F4BE6CCD";
    int len_real_key = sizeof(real_key) - 1; 
    unsigned char s[256];
    int i, j;
    for (i = 0; i < 256; i++) {
        s[i] = i;
    }
    j = 0;
    for (i = 0; i < 256; i++) {
        int key_idx = i % len_real_key;
        unsigned char k = real_key[key_idx] ^ 0x37;
        j = (j + s[i] + k) % 256;
        unsigned char temp = s[i];
        s[i] = s[j];
        s[j] = temp;
    }
    i = j = 0;
    unsigned char flag[sizeof(data)];
    int data_len = sizeof(data);
    for (int k = 0; k < data_len; k++) {
        i = (i + 1) % 256;
        j = (j + s[i]) % 256;
        unsigned char temp = s[i];
        s[i] = s[j];
        s[j] = temp;
        int t_idx = (s[i] + s[j]) % 256;
        unsigned char t = s[t_idx];
        t = (t << 4) | (t >> 4);
        flag[k] = data[k] ^ t;
    }
    printf("Flag: ");
    for (int k = 0; k < data_len; k++) {
        printf("%c", flag[k]);
    }
    printf("\n");
    return 0;
}
```
